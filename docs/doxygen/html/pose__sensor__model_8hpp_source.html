67  DynamicVector PredictMeasurement(const StateVector& state) const override {
68  Vector predicted_measurement = Vector::Zero();
69 
70  // Extract position from state
71  Eigen::Vector3d position = state.segment<3>(core::StateIndex::Position::X);
72 
73  // Extract orientation quaternion from state
74  Eigen::Quaterniond orientation(
75  state(core::StateIndex::Quaternion::W),
76  state(core::StateIndex::Quaternion::X),
77  state(core::StateIndex::Quaternion::Y),
78  state(core::StateIndex::Quaternion::Z)
79  );
80 
81  // Extract the sensor-to-body transform components
82  Eigen::Vector3d trans_b_s = sensor_pose_in_body_frame_.translation();
83  Eigen::Quaterniond rot_b_s(sensor_pose_in_body_frame_.rotation());
84 
85  // Compute predicted position in global frame
86  // p_sensor_global = p_body + R_body_to_global * p_sensor_in_body
87  Eigen::Vector3d predicted_position = position + orientation * trans_b_s;
88 
89  // Compute predicted orientation in global frame
90  // q_sensor_global = q_body * q_sensor_in_body
91  Eigen::Quaterniond predicted_orientation = orientation * rot_b_s;
92  predicted_orientation.normalize(); // Ensure unit quaternion
93 
94  // Fill measurement vector
95  predicted_measurement.segment<3>(MeasurementIndex::X) = predicted_position;
96  predicted_measurement(MeasurementIndex::QW) = predicted_orientation.w();
97  predicted_measurement(MeasurementIndex::QX) = predicted_orientation.x();
98  predicted_measurement(MeasurementIndex::QY) = predicted_orientation.y();
99  predicted_measurement(MeasurementIndex::QZ) = predicted_orientation.z();

110  DynamicJacobian GetMeasurementJacobian(const StateVector& /* state */) const override {
111  Jacobian jacobian = Jacobian::Zero();
112 
113  // Position part of the Jacobian - derivative with respect to position is identity
114  jacobian.block<3, 3>(MeasurementIndex::X, core::StateIndex::Position::X) =
115  Eigen::Matrix3d::Identity();
116 
117  // Remove coupling between linear and angular position in the jacobian by
118  // setting to zero. Derivation of partial of position with respect to the
119  // rotation in quaternion form can be found here:
120  // http://www.iri.upc.edu/people/jsola/JoanSola/objectes/notes/kinematics.pdf
121  jacobian.block<3, 4>(MeasurementIndex::X, core::StateIndex::Quaternion::W) =
122  Eigen::Matrix<double, 3, 4>::Zero();
123 
124  // Extract sensor-in-body quaternion components (q2)
125  const Eigen::Quaterniond q2(sensor_pose_in_body_frame_.rotation());
126  const double a2 = q2.w(), b2 = q2.x(), c2 = q2.y(), d2 = q2.z();
127 
128  // Quaternion Jacobian matrix - exactly matching the first provided formulation
129  Eigen::Matrix4d quaternion_product_jacobian;
130  quaternion_product_jacobian <<
131  // clang-format off
132  a2, b2, c2, d2,
133  -b2, a2, d2, -c2,
134  -c2, -d2, a2, b2,
135  -d2, c2, -b2, a2;
136  // clang-format on
137 
138  // Fill the quaternion part of the measurement Jacobian
139  jacobian.block<4, 4>(MeasurementIndex::QW, core::StateIndex::Quaternion::W) =
140  quaternion_product_jacobian;
141 
142  return jacobian;
143  }

183  const DynamicVector& measurement,
184  const StateFlags&,
185  StateVector& state,
186  StateCovariance& covariance) const override {
187  ValidateMeasurementSize(measurement);
188 
189  StateFlags initialized_states = StateFlags::Zero();
190 
191  // Extract position and orientation from measurement
192  Eigen::Vector3d sensor_position = measurement.segment<3>(MeasurementIndex::X);
193  Eigen::Quaterniond sensor_orientation(
194  measurement(MeasurementIndex::QW),
195  measurement(MeasurementIndex::QX),
196  measurement(MeasurementIndex::QY),
197  measurement(MeasurementIndex::QZ)
198  );
199 
200  // Extract the sensor-to-body transform components
201  Eigen::Vector3d trans_b_s = sensor_pose_in_body_frame_.translation();
202  Eigen::Quaterniond rot_b_s(sensor_pose_in_body_frame_.rotation());
203 
204  // Transform from sensor to body frame
205  // For orientation: q_body = q_sensor * (q_sensor_in_body)^-1
206  Eigen::Quaterniond body_orientation = sensor_orientation * rot_b_s.inverse();
207  body_orientation.normalize(); // Ensure unit quaternion
208 
209  // For position: p_body = p_sensor - R_body_to_global * p_sensor_in_body
210  Eigen::Vector3d body_position = sensor_position - body_orientation * trans_b_s;
211 
212  // Update state with initialized values
213  state.segment<3>(core::StateIndex::Position::Begin()) = body_position;
214  state(core::StateIndex::Quaternion::W) = body_orientation.w();
215  state(core::StateIndex::Quaternion::X) = body_orientation.x();
216  state(core::StateIndex::Quaternion::Y) = body_orientation.y();
217  state(core::StateIndex::Quaternion::Z) = body_orientation.z();
218 
219  // Transform measurement covariance to state covariance
220  // Note: This is a simplified approach; a more rigorous implementation would
221  // apply the proper uncertainty transformation through the Jacobian
222 
223  // For position covariance (direct mapping with lever arm effects)
224  Eigen::Matrix3d pos_cov = measurement_covariance_.block<3, 3>(0, 0);
225 
226  // For quaternion covariance (simplified transfer from measurement)
227  Eigen::Matrix4d quat_cov = measurement_covariance_.block<4, 4>(3, 3);
228 
229  // Update covariance blocks
230  covariance.block<3, 3>(
231  core::StateIndex::Position::Begin(),
232  core::StateIndex::Position::Begin()) = pos_cov;
233 
234  covariance.block<4, 4>(
235  core::StateIndex::Quaternion::Begin(),
236  core::StateIndex::Quaternion::Begin()) = quat_cov;
237 
238  // Mark states as initialized
239  initialized_states[core::StateIndex::Position::X] = true;
240  initialized_states[core::StateIndex::Position::Y] = true;
241  initialized_states[core::StateIndex::Position::Z] = true;
242  initialized_states[core::StateIndex::Quaternion::W] = true;
243  initialized_states[core::StateIndex::Quaternion::X] = true;
244  initialized_states[core::StateIndex::Quaternion::Y] = true;
