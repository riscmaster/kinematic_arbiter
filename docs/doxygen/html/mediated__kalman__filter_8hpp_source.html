114  bool ProcessMeasurementByIndex(size_t sensor_index,
115  const MeasurementModelInterface::DynamicVector& measurement,
116  double timestamp) {
117  if (sensor_index >= sensors_.size()) {
118  std::cerr << "Warning: ProcessMeasurementByIndex called with invalid sensor index: " << sensor_index << std::endl;
119  return false;
120  }
121 
122  auto sensor = sensors_[sensor_index];
123 
124  // Set reference time if not initialized
125  if (reference_time_ == std::numeric_limits<double>::lowest()) {
126  reference_time_ = timestamp;
127  }
128 
129  // Quick validation of measurement and timestamp before any expensive operations
130  if (!sensor->ValidateMeasurementAndTime(measurement, timestamp, reference_time_, max_delay_window_)) {
131  return false;
132  }
133 
134  // Try to initialize uninitiated states
135  StateFlags initializable = sensor->GetInitializableStates();
136  StateFlags not_initialized = initialized_states_.select(StateFlags::Zero(), StateFlags::Ones());
137  StateFlags uninit_states = initializable.cwiseProduct(not_initialized);
138 
139  // Only attempt initialization for states we haven't initialized yet
140  if (uninit_states.any()) {
141  StateFlags new_states = sensor->InitializeState(
142  measurement, initialized_states_, reference_state_, reference_covariance_);
143  initialized_states_ = initialized_states_.cwiseMax(new_states);
144  }
145 
146  double dt = timestamp - reference_time_;
147 
148  // Predict state to measurement time
149  StateMatrix A = process_model_->GetTransitionMatrix(reference_state_, dt);
150  StateVector state_at_sensor_time;
151 
152 
153  state_at_sensor_time = process_model_->PredictState(reference_state_, dt);
154 
155 
156  // Validate predicted state
157  if (!state_at_sensor_time.allFinite()) {
158  std::cerr << "Predicted state contains NaN/Inf values at time " << timestamp << std::endl;
159  return false;
160  }
161 
162  StateMatrix Q = process_model_->GetProcessNoiseCovariance(dt);
163  StateMatrix covariance_at_sensor_time = A * reference_covariance_ * A.transpose() + Q;
164  covariance_at_sensor_time = 0.5 * (covariance_at_sensor_time + covariance_at_sensor_time.transpose());
165 
166  // Compute auxiliary data
167  MeasurementModelInterface::MeasurementAuxData aux_data;
168 
169  // Validate and mediate the measurement
170  bool measurement_valid = sensor->ValidateAndMediate(
171  state_at_sensor_time, covariance_at_sensor_time, timestamp, measurement, aux_data);
172 
173  if (!measurement_valid) {
174  return false;
175  }
176 
177  Eigen::MatrixXd PHt = covariance_at_sensor_time * aux_data.jacobian.transpose();
178  Eigen::MatrixXd K_transpose = aux_data.innovation_covariance.ldlt().solve(PHt.transpose());
179  Eigen::MatrixXd kalman_gain = K_transpose.transpose();
180 
181  // Update state estimate
182  StateVector updated_state = state_at_sensor_time + kalman_gain * aux_data.innovation;
183 
184  // Update covariance (Joseph form for numerical stability)
185  StateMatrix I_KH = StateMatrix::Identity() - kalman_gain * aux_data.jacobian;
186  StateMatrix updated_covariance = I_KH * covariance_at_sensor_time * I_KH.transpose() +
187  kalman_gain * sensor->GetMeasurementCovariance() * kalman_gain.transpose();
188 
189  updated_covariance = 0.5 * (updated_covariance + updated_covariance.transpose());
190 
191  if (!updated_covariance.allFinite()) {
192  std::cerr << "Updated covariance contains NaN/Inf values for at time " << timestamp << std::endl;
193  return false;
194  }
195 
196  // Update process noise
197  process_model_->UpdateProcessNoise(
198  state_at_sensor_time,
199  updated_state,
200  sensor->GetValidationParams().process_to_measurement_noise_ratio,
201  dt
202  );
203 
204  // Update reference state based on timestamp
205  if (timestamp > reference_time_) {
206  reference_state_ = updated_state;
207  reference_covariance_ = updated_covariance;
208  reference_time_ = timestamp;
209  } else {
210  double forward_dt = reference_time_ - timestamp;
211  reference_state_ = process_model_->PredictState(updated_state, forward_dt);
212 
213  if (!reference_state_.allFinite()) {
214  std::cerr << "Forward propagation produced invalid state for at time " << timestamp << std::endl;
215  return false;
216  }
217 
218  StateMatrix A_forward = process_model_->GetTransitionMatrix(updated_state, forward_dt);
219  StateMatrix Q_forward = process_model_->GetProcessNoiseCovariance(forward_dt);
220  reference_covariance_ = A_forward * updated_covariance * A_forward.transpose() + Q_forward;
221  reference_covariance_ = 0.5 * (reference_covariance_ + reference_covariance_.transpose());
222  }
223 
224  return true;
225  }
