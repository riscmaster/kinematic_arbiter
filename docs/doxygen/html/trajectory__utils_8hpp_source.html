31 Eigen::Matrix<double, core::StateIndex::kFullStateSize, 1> Figure8Trajectory(
32  double time,
33  const Figure8Config& config = Figure8Config()) {
34 
35  using SIdx = core::StateIndex;
36  using StateVector = Eigen::Matrix<double, SIdx::kFullStateSize, 1>;
37 
38  // Trajectory parameters
39  const double a = config.length * 0.5; // x amplitude
40  const double b = config.width * 0.5; // y amplitude
41  const double c = config.width * std::tan(config.width_slope); // z amplitude
42 
43  // Period and frequency calculations
44  const double T = M_PI * std::sqrt(a*a + 4*(b*b + c*c)) / config.max_velocity;
45  const double w1 = 2 * M_PI / T; // y,z frequency
46  const double w2 = w1 * 0.5; // x frequency
47  const double t = time; // current time
48 
49  // Create state vector
50  StateVector states = StateVector::Zero();
51 
52  // Position vector in world frame
53  Eigen::Vector3d position(
54  a * std::cos(w2 * t),
55  b * std::sin(w1 * t),
56  c * std::sin(w1 * t));
57 
58  // Store position
59  states.segment<3>(SIdx::Position::Begin()) = position;
60 
61  // Linear Inertial (World Frame) Velocity
62  Eigen::Vector3d i_v(
63  -a * w2 * std::sin(w2 * t),
64  b * w1 * std::cos(w1 * t),
65  c * w1 * std::cos(w1 * t));
66 
67  // Linear Inertial (World Frame) Acceleration
68  Eigen::Vector3d i_a(
69  -a * w2 * w2 * std::cos(w2 * t),
70  -b * w1 * w1 * std::sin(w1 * t),
71  -c * w1 * w1 * std::sin(w1 * t));
72 
73  // Linear Inertial (World Frame) Jerk
74  Eigen::Vector3d i_j(
75  a * w2 * w2 * w2 * std::sin(w2 * t),
76  -b * w1 * w1 * w1 * std::cos(w1 * t),
77  -c * w1 * w1 * w1 * std::cos(w1 * t));
78 
79  // Current Orientation (body X aligned with velocity)
80  Eigen::Quaterniond orientation;
81  orientation.setFromTwoVectors(Eigen::Vector3d::UnitX(), i_v.normalized());
82  orientation.normalize();
83 
84  // Store quaternion
85  states[SIdx::Quaternion::W] = orientation.w();
86  states[SIdx::Quaternion::X] = orientation.x();
87  states[SIdx::Quaternion::Y] = orientation.y();
88  states[SIdx::Quaternion::Z] = orientation.z();
89 
90  // Convert world velocity to body frame
91  Eigen::Vector3d b_v = orientation.inverse() * i_v;
92  states.segment<3>(SIdx::LinearVelocity::Begin()) = b_v;
93 
94  // 1. Compute angular velocity in world frame
95  Eigen::Vector3d omega_world;
96  if (i_v.squaredNorm() > 1e-9) {
97  omega_world = i_v.cross(i_a) / i_v.squaredNorm();
98  } else {
99  omega_world.setZero();
100  }
101 
102  // 2. Transform to body frame
103  Eigen::Vector3d omega_body = orientation.inverse() * omega_world;
104  states.segment<3>(SIdx::AngularVelocity::Begin()) = omega_body;
105 
106  // 3. Compute body-frame acceleration PROPERLY
107  Eigen::Vector3d b_a = orientation.inverse() * i_a;
108  b_a -= 2 * omega_body.cross(b_v); // Full Coriolis compensation
109  states.segment<3>(SIdx::LinearAcceleration::Begin()) = b_a;
110 
111  // 4. Angular acceleration (derivative of omega_world)
112  Eigen::Vector3d alpha_world;
113  if (i_v.squaredNorm() > 1e-9) {
114  alpha_world = (i_v.cross(i_j) * i_v.squaredNorm() -
115  2 * i_v.cross(i_a) * i_v.dot(i_a))
116  / (i_v.squaredNorm() * i_v.squaredNorm());
117  } else {
118  alpha_world.setZero();
119  }
120 
121  // 5. Transform to body frame
122  Eigen::Vector3d alpha_body = orientation.inverse() * alpha_world;
123  states.segment<3>(SIdx::AngularAcceleration::Begin()) = alpha_body;
124 
125  // 6. Extract and store roll rate (rotation about body X axis)
126  double roll_rate = omega_body.dot(Eigen::Vector3d::UnitX());
127  states[SIdx::AngularVelocity::X] = roll_rate;
128 
129  // 7. Extract and store roll acceleration (about body X axis)
130  double roll_accel = alpha_body.dot(Eigen::Vector3d::UnitX());
131  states[SIdx::AngularAcceleration::X] = roll_accel;
132 
133  return states;
134 }
