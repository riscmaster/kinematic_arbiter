67  DynamicVector PredictMeasurement(const StateVector& state) const override {
68  Vector predicted_measurement = Vector::Zero();
69 
70  // Extract linear velocity from state
71  Eigen::Vector3d body_lin_vel = state.segment<3>(core::StateIndex::LinearVelocity::X);
72 
73  // Extract angular velocity from state
74  Eigen::Vector3d body_ang_vel = state.segment<3>(core::StateIndex::AngularVelocity::X);
75 
76  // Extract the sensor-to-body transform components
77  Eigen::Vector3d trans_b_s = sensor_pose_in_body_frame_.translation();
78  Eigen::Matrix3d rot_b_s = sensor_pose_in_body_frame_.rotation();
79 
80  // Calculate the skew-symmetric matrix for the cross product: ω × r
81  auto skew_matrix = [](const Eigen::Vector3d& v) -> Eigen::Matrix3d {
82  Eigen::Matrix3d skew;
83  skew << 0, -v(2), v(1),
84  v(2), 0, -v(0),
85  -v(1), v(0), 0;
86  return skew;
87  };
88 
89  // Calculate linear velocity at the sensor location in body frame
90  // v_sensor = v_body + ω × r, where r is the position of the sensor in the body frame
91  Eigen::Vector3d body_sensor_lin_vel = body_lin_vel + skew_matrix(body_ang_vel) * trans_b_s;
92 
93  // Transform velocities to sensor frame
94  Eigen::Vector3d sensor_lin_vel = rot_b_s.transpose() * body_sensor_lin_vel;
95  Eigen::Vector3d sensor_ang_vel = rot_b_s.transpose() * body_ang_vel;
96 
97  // Fill measurement vector
98  predicted_measurement.segment<3>(MeasurementIndex::VX) = sensor_lin_vel;
99  predicted_measurement.segment<3>(MeasurementIndex::WX) = sensor_ang_vel;
100 
101  return predicted_measurement;
102  }

110  DynamicJacobian GetMeasurementJacobian(const StateVector&) const override {
111  Jacobian jacobian = Jacobian::Zero();
112 
113  // Extract transform parameters
114  Eigen::Vector3d sensor_offset = sensor_pose_in_body_frame_.translation();
115  Eigen::Matrix3d body_to_sensor_rotation = sensor_pose_in_body_frame_.rotation().transpose();
116 
117  // ===== LINEAR VELOCITY JACOBIAN =====
118 
119  // Linear velocity w.r.t. linear velocity - simple rotation from body to sensor frame
120  jacobian.block<3, 3>(MeasurementIndex::VX, core::StateIndex::LinearVelocity::X) = body_to_sensor_rotation;
121 
122  // Linear velocity w.r.t. angular velocity - lever arm effect
123  // The lever arm effect can be represented by partial derivatives of the cross product
124  // These derivatives have a standard form for cross products
125 
126  // Define cross product derivatives in a more compact way
127  // Following standard notation where [r×] is the skew-symmetric matrix of vector r
128  // For a sensor at position r, the derivatives of ω×r with respect to each ω component
129  // correspond to the columns of a skew matrix of r
130 
131  // Create skew-symmetric matrix of the sensor offset
132  Eigen::Matrix3d skew_offset = Eigen::Matrix3d::Zero();
133  skew_offset <<
134  0, -sensor_offset.z(), sensor_offset.y(),
135  sensor_offset.z(), 0, -sensor_offset.x(),
136  -sensor_offset.y(), sensor_offset.x(), 0;
137 
138  // Each column of this matrix represents the lever arm effect for one angular velocity component
139  // Extract columns for better readability
140  Eigen::Vector3d lever_arm_effect_wx = -skew_offset.col(0); // Effect of ω_x
141  Eigen::Vector3d lever_arm_effect_wy = -skew_offset.col(1); // Effect of ω_y
142  Eigen::Vector3d lever_arm_effect_wz = -skew_offset.col(2); // Effect of ω_z
143 
144  // Apply body-to-sensor rotation and fill Jacobian
145  jacobian.block<3, 1>(MeasurementIndex::VX, core::StateIndex::AngularVelocity::X) =
146  body_to_sensor_rotation * lever_arm_effect_wx;
147  jacobian.block<3, 1>(MeasurementIndex::VX, core::StateIndex::AngularVelocity::Y) =
148  body_to_sensor_rotation * lever_arm_effect_wy;
149  jacobian.block<3, 1>(MeasurementIndex::VX, core::StateIndex::AngularVelocity::Z) =
150  body_to_sensor_rotation * lever_arm_effect_wz;
151 
152  // ===== ANGULAR VELOCITY JACOBIAN =====
153 
154  // Angular velocity w.r.t. angular velocity - simple rotation from body to sensor frame
155  jacobian.block<3, 3>(MeasurementIndex::WX, core::StateIndex::AngularVelocity::X) = body_to_sensor_rotation;
156 
157  return jacobian;
158  }

194  const DynamicVector& measurement,

198  ValidateMeasurementSize(measurement);
199 
200  StateFlags initialized_states = StateFlags::Zero();
201 
202  // Extract measurement components in sensor frame
203  const Eigen::Vector3d sensor_lin_vel = measurement.segment<3>(0);
204  const Eigen::Vector3d sensor_ang_vel = measurement.segment<3>(3);
205 
206  // Extract the sensor-to-body transform components
207  const Eigen::Vector3d trans_b_s = sensor_pose_in_body_frame_.translation();
208  const Eigen::Matrix3d rot_b_s = sensor_pose_in_body_frame_.rotation();
209 
210  // Calculate the skew-symmetric matrix for the cross product: ω × r
211  auto skew_matrix = [](const Eigen::Vector3d& v) -> Eigen::Matrix3d {
212  Eigen::Matrix3d skew;
213  skew << 0, -v(2), v(1),
214  v(2), 0, -v(0),
215  -v(1), v(0), 0;
216  return skew;
217  };
218 
219  // First, compute body angular velocity
220  const Eigen::Vector3d body_ang_vel = rot_b_s * sensor_ang_vel;
221 
222  // Convert sensor linear velocity to body frame
223  const Eigen::Vector3d sensor_lin_vel_in_body = rot_b_s * sensor_lin_vel;
224 
225  // Account for the lever arm effect
226  const Eigen::Vector3d lever_arm_effect = skew_matrix(body_ang_vel) * trans_b_s;
227 
228  // Compute body linear velocity
229  const Eigen::Vector3d body_lin_vel = sensor_lin_vel_in_body - lever_arm_effect;
230 
231  // Initialize state vector
232  state.segment<3>(core::StateIndex::LinearVelocity::Begin()) = body_lin_vel;
233  state.segment<3>(core::StateIndex::AngularVelocity::Begin()) = body_ang_vel;
234 
235  // Set all velocity states as initialized
236  initialized_states[core::StateIndex::LinearVelocity::X] = true;
237  initialized_states[core::StateIndex::LinearVelocity::Y] = true;
238  initialized_states[core::StateIndex::LinearVelocity::Z] = true;
239  initialized_states[core::StateIndex::AngularVelocity::X] = true;
240  initialized_states[core::StateIndex::AngularVelocity::Y] = true;
241  initialized_states[core::StateIndex::AngularVelocity::Z] = true;
242 
243  // Transform covariance from sensor to body frame
244  // For angular velocity (straightforward rotation)
245  Eigen::Matrix3d ang_vel_cov = rot_b_s *
246  measurement_covariance_.block<3, 3>(3, 3) *
247  rot_b_s.transpose();
248 
249  // For linear velocity (approximate - ignoring cross-terms with angular velocity)
250  Eigen::Matrix3d lin_vel_cov = rot_b_s *
251  measurement_covariance_.block<3, 3>(0, 0) *
252  rot_b_s.transpose();
253 
254  // Set covariance blocks in state covariance
255  covariance.block<3, 3>(
256  core::StateIndex::LinearVelocity::Begin(),
257  core::StateIndex::LinearVelocity::Begin()) = lin_vel_cov;
258 
259  covariance.block<3, 3>(
260  core::StateIndex::AngularVelocity::Begin(),
261  core::StateIndex::AngularVelocity::Begin()) = ang_vel_cov;
262 
263 
264  return initialized_states;
265  }
