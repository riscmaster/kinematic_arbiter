36  // Extract state components
37  const Eigen::Quaterniond q(

39  state(StateIndex::Quaternion::X),
40  state(StateIndex::Quaternion::Y),

42  );
43  const Eigen::Vector3d omega = state.segment<3>(StateIndex::AngularVelocity::X);
44  const Eigen::Vector3d a_linear = state.segment<3>(StateIndex::LinearAcceleration::X);
45  const Eigen::Vector3d alpha = state.segment<3>(StateIndex::AngularAcceleration::X);
46 
47  // Extract sensor configuration
48  const Eigen::Vector3d& r = sensor_pose_in_body_frame_.translation();
49  const Eigen::Matrix3d& R_BS = sensor_pose_in_body_frame_.rotation().transpose();
50 
51  // Gravity vector in world frame
52  const Eigen::Vector3d g_W(0.0, 0.0, kGravity);
53 
54  // Base gyroscope measurement without bias
55  Eigen::Vector3d gyro = R_BS * omega;
56 
57  // Base accelerometer measurement without bias
58  Eigen::Vector3d accel = R_BS * (
59  a_linear +
60  q.inverse() * g_W +
61  alpha.cross(r) +
62  omega.cross(omega.cross(r))
63  );
64

68  accel += bias_estimator_.GetAccelBias();
69  }
70

73  predicted_measurement.segment<3>(MeasurementIndex::AX) = accel;

101 ImuSensorModel::Base::DynamicJacobian ImuSensorModel::GetMeasurementJacobian(
102  const StateVector& state) const {
103  Jacobian jacobian = Jacobian::Zero();
104 
105  // Extract state components and sensor configuration
106  const Eigen::Vector3d& r = sensor_pose_in_body_frame_.translation();
107  const Eigen::Matrix3d& R_BS = sensor_pose_in_body_frame_.rotation().transpose();
108 
109  const Eigen::Vector3d omega = state.segment<3>(StateIndex::AngularVelocity::X);
110 
111  // Extract and normalize quaternion orientation
112  Eigen::Quaterniond q(
113  state(StateIndex::Quaternion::W),
114  state(StateIndex::Quaternion::X),
115  state(StateIndex::Quaternion::Y),
116  state(StateIndex::Quaternion::Z)
117  );
118  q.normalize();
119 
120  // Helper function to create a skew-symmetric matrix for cross products
121  auto skew = [](const Eigen::Vector3d& v) -> Eigen::Matrix3d {
122  return (Eigen::Matrix3d() <<
123  0, -v.z(), v.y(),
124  v.z(), 0, -v.x(),
125  -v.y(), v.x(), 0).finished();
126  };
127 
128  // ==== LOWER BLOCK: GYROSCOPE JACOBIAN ====
129  // Gyro measurement is simply body angular velocity rotated to sensor frame
130  jacobian.block<3, 3>(MeasurementIndex::GX, StateIndex::AngularVelocity::X) = R_BS;
131 
132  // ==== UPPER BLOCK: ACCELEROMETER JACOBIAN ====
133 
134  // 1. Linear acceleration direct mapping
135  jacobian.block<3, 3>(MeasurementIndex::AX, StateIndex::LinearAcceleration::X) = R_BS;
136 
137  // 2. Angular acceleration effect (tangential acceleration): -R_{BS}[r]_\times
138  jacobian.block<3, 3>(MeasurementIndex::AX, StateIndex::AngularAcceleration::X) =
139  -R_BS * skew(r);
140 
141  // 3. Angular velocity effect on acceleration
142  // To fix the issues with the angular velocity Jacobian, we'll use the expanded cross product formula
143  Eigen::Matrix3d omega_cross_r_jacobian = Eigen::Matrix3d::Zero();
144 
145  // Manually construct the Jacobian for d(ω×(ω×r))/dω
146  // This accounts for both terms in the derivative
147  for (int i = 0; i < 3; i++) {
148  Eigen::Vector3d e_i = Eigen::Vector3d::Zero();
149  e_i(i) = 1.0;
150 
151  // First term: e_i × (ω × r)
152  Eigen::Vector3d term1 = e_i.cross(omega.cross(r));
153 
154  // Second term: ω × (e_i × r)
155  Eigen::Vector3d term2 = omega.cross(e_i.cross(r));
156 
157  // Combined effect
158  omega_cross_r_jacobian.col(i) = term1 + term2;
159  }
160 
161  jacobian.block<3, 3>(MeasurementIndex::AX, StateIndex::AngularVelocity::X) =
162  R_BS * omega_cross_r_jacobian;
163 
164  // 4. Quaternion effect on gravity
165  // Derive Jacobian for rotation of gravity vector by quaternion
166  // For gravity vector g_W = [0, 0, g]^T in world frame
167  // The rotated gravity in body frame is: g_B = q.inverse() * g_W * q
168  const double qw = q.w(), qx = q.x(), qy = q.y(), qz = q.z();
169  const double g = kGravity;
170 
171  // Using the derivation from section 4.3 of the documentation:
172  // For g_W = [0, 0, g]^T, the rotation q.inverse() * g_W gives:
173  // g_B = g * [
174  // 2(q_x*q_z - q_w*q_y)
175  // 2(q_y*q_z + q_w*q_x)
176  // 1 - 2(q_x^2 + q_y^2)
177  // ]
178  //
179  // The Jacobian ∂g_B/∂q is therefore:
180  Eigen::Matrix<double, 3, 4> quaternion_gravity_jacobian;
181  quaternion_gravity_jacobian <<
182  /* ∂/∂q_w */ /* ∂/∂q_x */ /* ∂/∂q_y */ /* ∂/∂q_z */
183  -2*g*qy, 2*g*qz, -2*g*qw, 2*g*qx, // ∂g_x/∂q
184  2*g*qx, 2*g*qw, 2*g*qz, 2*g*qy, // ∂g_y/∂q
185  0, -4*g*qx, -4*g*qy, 0; // ∂g_z/∂q
186 
187  // Apply sensor-to-body rotation to transform the Jacobian to sensor frame
188  jacobian.block<3, 4>(MeasurementIndex::AX, StateIndex::Quaternion::W) =
189  R_BS * quaternion_gravity_jacobian;
190 
191  return jacobian;
192 }

285 bool ImuSensorModel::IsStationary(
286  const StateVector& state,
287  const StateCovariance& state_covariance,
288  const Vector& measurement) const {
289 
290  // Extract measurements directly using segment
291  Eigen::Vector3d measured_accel = measurement.segment<3>(MeasurementIndex::AX);
292 
293  // Compute acceleration norm (should be close to gravity when stationary)
294  double accel_norm = measured_accel.norm();
295 
296  // Pre-compute critical values for different degrees of freedom
297  double critical_value6dof = utils::CalculateChiSquareCriticalValueNDof(5, config_.stationary_confidence_threshold);
298  double critical_value1dof = utils::CalculateChiSquareCriticalValueNDof(0, config_.stationary_confidence_threshold);
299 
300  // 1. Test state velocities (6-DOF)
301  int state_vel_idx = StateIndex::LinearVelocity::X;
302  Eigen::Matrix<double, 6, 1> state_vel = state.segment<6>(state_vel_idx);
303  Eigen::Matrix<double, 6, 6> state_vel_cov = state_covariance.block<6, 6>(state_vel_idx, state_vel_idx);
304  bool is_state_vel0 = state_vel.dot(state_vel_cov.llt().solve(state_vel)) < critical_value6dof;
305 
306  // 2. Test state accelerations (6-DOF)
307  int state_acc_idx = StateIndex::LinearAcceleration::X;
308  Eigen::Matrix<double, 6, 1> state_acc = state.segment<6>(state_acc_idx);
309  Eigen::Matrix<double, 6, 6> state_acc_cov = state_covariance.block<6, 6>(state_acc_idx, state_acc_idx);
310  bool is_state_acc0 = state_acc.dot(state_acc_cov.llt().solve(state_acc)) < critical_value6dof;
311 
312  // 3. Test IMU acceleration norm against gravity (1-DOF)
313  double gravity_diff = accel_norm - kGravity;
314  double accel_norm_var = this->measurement_covariance_.block<3, 3>(MeasurementIndex::AX, MeasurementIndex::AX).trace();
315  double gravity_mahalanobis = (gravity_diff * gravity_diff) / (accel_norm_var + kGravityVariance);
316  bool is_imu_accel_gravity = gravity_mahalanobis < critical_value1dof;
317 
318  // Ignoring angular velocity from IMU for now due to confounding bias effects
319 
320  return is_state_vel0 && is_state_acc0 && is_imu_accel_gravity;
321 }

371 typename ImuSensorModel::StateFlags ImuSensorModel::InitializeState(
372  const ImuSensorModel::Base::DynamicVector& measurement,
373  const StateFlags& ,
374  StateVector& state,
375  StateCovariance& covariance) const {
376 
377  ValidateMeasurementSize(measurement);
378 
379  StateFlags initialized_states = StateFlags::Zero();
380 
381  // Extract gyro and accelerometer measurements
382  Eigen::Vector3d gyro = measurement.segment<3>(0);
383  Eigen::Vector3d accel = measurement.segment<3>(3);
384 
385  // Transform to body frame
386  const Eigen::Matrix3d R_SB = sensor_pose_in_body_frame_.rotation();
387 
388  // Apply bias correction if calibration is enabled
389  if (config_.calibration_enabled) {
390  gyro -= bias_estimator_.GetGyroBias();
391  accel -= bias_estimator_.GetAccelBias();
392  }
393 
394  // Transform to body frame
395  Eigen::Vector3d omega_body = R_SB * gyro;
396  Eigen::Vector3d accel_body = R_SB * accel;
397 
398  // Initialize angular velocity
399  state.segment<3>(StateIndex::AngularVelocity::Begin()) = omega_body;
400 
401  // Set angular velocity covariance
402  Eigen::Matrix3d gyro_cov = measurement_covariance_.block<3, 3>(0, 0);
403  Eigen::Matrix3d omega_body_cov = R_SB * gyro_cov * R_SB.transpose();
404  covariance.block<3, 3>(
405  StateIndex::AngularVelocity::Begin(),
406  StateIndex::AngularVelocity::Begin()) = omega_body_cov;
407 
408  // Mark angular velocity as initialized
409  initialized_states[StateIndex::AngularVelocity::X] = true;
410  initialized_states[StateIndex::AngularVelocity::Y] = true;
411  initialized_states[StateIndex::AngularVelocity::Z] = true;
412 
413  // Calculate accelerometer covariance in body frame
414  Eigen::Matrix3d accel_cov = measurement_covariance_.block<3, 3>(3, 3);
415  Eigen::Matrix3d accel_body_cov = R_SB * accel_cov * R_SB.transpose();
416 
417  // Check if IMU is stationary for orientation initialization
418  bool is_stationary = IsStationary(state, covariance, measurement);
419 
420  // Always extract current yaw regardless of validity
421  // Extract current quaternion
422  Eigen::Quaterniond current_q(
423  state(StateIndex::Quaternion::W),
424  state(StateIndex::Quaternion::X),
425  state(StateIndex::Quaternion::Y),
426  state(StateIndex::Quaternion::Z)
427  );
428  current_q.normalize();
429 
430  // Convert to rotation matrix and extract yaw
431  Eigen::Matrix3d rot_matrix = current_q.toRotationMatrix();
432  double current_yaw = std::atan2(rot_matrix(1, 0), rot_matrix(0, 0));
433  if (is_stationary || (accel_body.norm() > 0.5 * kGravity && accel_body.norm() < 1.5 * kGravity)) {
434  double ay = accel_body(1)/accel_body.norm();
435  double az = accel_body(2)/accel_body.norm();
436  double roll = std::atan2(ay, az);
437  // Calculate pitch from accelerometer when stationary
438  double ax = accel_body(0);
439  double pitch = std::asin(-ax / accel_body.norm());
440  double roll_pitch_variance = (M_PI/6.0) * (M_PI/6.0);
441 
442  if (is_stationary) {
443  // Calculate roll from accelerometer when stationary
444  ay = accel_body(1);
445  az = accel_body(2);
446  roll = std::atan2(ay, az);
447 
448  // Propagate uncertainty for roll
449  double d_roll_d_ay = std::abs(az / (ay*ay + az*az));
450  double d_roll_d_az = std::abs(-ay / (ay*ay + az*az));
451  Eigen::Vector2d d_roll_d_ayz;
452  d_roll_d_ayz << d_roll_d_ay, d_roll_d_az;
453  double roll_var = d_roll_d_ayz.transpose() *
454  accel_body_cov.block<2,2>(1,1) *
455  d_roll_d_ayz;
456 
457  // Calculate pitch from accelerometer when stationary
458  ax = accel_body(0);
459  pitch = std::asin(-ax / kGravity);
460 
461  // Propagate uncertainty for pitch
462  double d_pitch_d_ax = std::abs(1.0 / (kGravity * std::sqrt(1.0 - (ax*ax)/(kGravity*kGravity))));
463  double pitch_var = d_pitch_d_ax * accel_body_cov(0,0) * d_pitch_d_ax;
464 
465  // Set appropriate covariance for quaternion components
466  roll_pitch_variance = std::max(roll_var, pitch_var);
467  }
468 
469  // Create rotation matrix from roll, pitch, and yaw
470  Eigen::Matrix3d roll_rot = Eigen::AngleAxisd(roll, Eigen::Vector3d::UnitX()).toRotationMatrix();
471  Eigen::Matrix3d pitch_rot = Eigen::AngleAxisd(pitch, Eigen::Vector3d::UnitY()).toRotationMatrix();
472  Eigen::Matrix3d yaw_rot = Eigen::AngleAxisd(current_yaw, Eigen::Vector3d::UnitZ()).toRotationMatrix();
473 
474  // Combine rotations in XYZ order (roll-pitch-yaw)
475  rot_matrix = roll_rot * pitch_rot * yaw_rot;
476  // Convert to quaternion
477  Eigen::Quaterniond q(rot_matrix);
478  q.normalize();
479 
480  // Update state with quaternion
481  state(StateIndex::Quaternion::W) = q.w();
482  state(StateIndex::Quaternion::X) = q.x();
483  state(StateIndex::Quaternion::Y) = q.y();
484  state(StateIndex::Quaternion::Z) = q.z();
485 
486 
487  // Create a mask matrix for yaw-related components (W and Z)
488  Eigen::Matrix4d yaw_mask = Eigen::Matrix4d::Zero();
489  yaw_mask(0, 0) = 1.0; // W component
490  yaw_mask(3, 3) = 1.0; // Z component
491 
492  // Create a mask for roll/pitch components (W, X, Y)
493  Eigen::Matrix4d roll_pitch_mask = Eigen::Matrix4d::Identity() - yaw_mask;
494 
495  // Combine the covariances:
496  // - For roll/pitch components: use the newly calculated variance
497  // - For yaw components: preserve the original covariance magnitude
498  covariance.block<4, 4>(StateIndex::Quaternion::Begin(), StateIndex::Quaternion::Begin()) =
499  (roll_pitch_mask * roll_pitch_variance) +
500  (yaw_mask * covariance.block<4, 4>(StateIndex::Quaternion::Begin(), StateIndex::Quaternion::Begin()).norm());
501 
502  // Mark roll and pitch as initialized (quaternion components)
503  initialized_states[StateIndex::Quaternion::W] = true;
504  initialized_states[StateIndex::Quaternion::X] = true;
505  initialized_states[StateIndex::Quaternion::Y] = true;
506 
507  // When stationary, linear acceleration should be zero in inertial frame
508  // and angular acceleration should be zero
509  state.segment<3>(StateIndex::LinearAcceleration::Begin()).setZero();
510  state.segment<3>(StateIndex::AngularAcceleration::Begin()).setZero();
511 
512  // Set covariance for linear and angular acceleration
513  covariance.block<3, 3>(
514  StateIndex::LinearAcceleration::Begin(),
515  StateIndex::LinearAcceleration::Begin()) = accel_body_cov;
516 
517  covariance.block<3, 3>(
518  StateIndex::AngularAcceleration::Begin(),
519  StateIndex::AngularAcceleration::Begin()) = omega_body_cov;
520 
521  // Mark linear and angular acceleration as initialized
522  for (int i = 0; i < 3; i++) {
523  initialized_states[StateIndex::LinearAcceleration::Begin() + i] = true;
524  initialized_states[StateIndex::AngularAcceleration::Begin() + i] = true;
525  }
526  }
527 
528  return initialized_states;
529 }
