55  StateVector PredictState(const StateVector& current_state, double time_step) const override {
56  StateVector new_states = current_state;
57 
58  if (fabs(time_step) <= std::numeric_limits<double>::epsilon()) {
59  return new_states;
60  }
61 
62  // Extract quaternion and normalize
63  Eigen::Quaterniond orientation(
64  current_state[SIdx::Quaternion::W],
65  current_state[SIdx::Quaternion::X],
66  current_state[SIdx::Quaternion::Y],
67  current_state[SIdx::Quaternion::Z]
68  );
69  orientation.normalize();
70 
71  // Get rotation matrix
72  Matrix3d rotation_matrix_b_to_w = orientation.toRotationMatrix();
73 
74  // Extract velocity and acceleration components
75  const Vector3d linear_velocity =
76  current_state.segment<3>(SIdx::LinearVelocity::Begin());
77 
78  const Vector3d angular_velocity =
79  current_state.segment<3>(SIdx::AngularVelocity::Begin());
80 
81  const Vector3d linear_acceleration =
82  current_state.segment<3>(SIdx::LinearAcceleration::Begin());
83 
84  const Vector3d angular_acceleration =
85  current_state.segment<3>(SIdx::AngularAcceleration::Begin());
86 
87  // Linear XYZ Position Prediction Model
88  new_states.segment<3>(SIdx::Position::Begin()) +=
89  rotation_matrix_b_to_w * ((linear_velocity + time_step * 0.5 * linear_acceleration) * time_step);
90 
91  // Angular Position (Quaternion) Prediction Model
92  // Use quaternion kinematics with angular acceleration: q(t+dt) = exp(0.5*ω*dt + 0.25*ω̇*dt²) * q(t)
93  Eigen::Quaterniond delta_q;
94 
95  // Calculate combined angular motion vector v = 0.5*ω*dt + 0.25*ω̇*dt²
96  // This is the same vector used in the quaternion update: q(t+dt) = exp(v) * q(t)
97  double t = time_step;
98  Vector3d combined_angular_motion = angular_velocity * t + 0.5 * angular_acceleration * t * t;
99  double motion_norm = combined_angular_motion.norm();
100 
101  if (motion_norm > std::numeric_limits<double>::min()) {
102  Vector3d axis = combined_angular_motion / motion_norm;
103  double angle = motion_norm;
104  delta_q = Eigen::Quaterniond(Eigen::AngleAxisd(angle, axis));
105  } else {
106  delta_q = Eigen::Quaterniond::Identity();
107  }
108 
109  Eigen::Quaterniond new_quaternion = delta_q * orientation;
110  new_quaternion.normalize();
111 
112  new_states[SIdx::Quaternion::W] = new_quaternion.w();
113  new_states[SIdx::Quaternion::X] = new_quaternion.x();
114  new_states[SIdx::Quaternion::Y] = new_quaternion.y();
115  new_states[SIdx::Quaternion::Z] = new_quaternion.z();
116 
117  // Velocity prediction (integrate acceleration)
118  new_states.segment<3>(SIdx::LinearVelocity::Begin()) += time_step * linear_acceleration;
119  new_states.segment<3>(SIdx::AngularVelocity::Begin()) += time_step * angular_acceleration;
120  // Acceleration is modelled as exponential decay: a(t+dt) = a(t) * exp(-lambda * dt)
121  // With lambda = 46.05, acceleration decays to near zero within 0.1s
122  new_states.segment<3>(SIdx::LinearAcceleration::Begin()) =
123  linear_acceleration * std::exp(-kLambda * time_step);
124  new_states.segment<3>(SIdx::AngularAcceleration::Begin()) =
125  angular_acceleration * std::exp(-kLambda * time_step);
126 
127  return new_states;
128  }

140  StateMatrix GetTransitionMatrix(const StateVector& current_state, double time_step) const override {
141  if (fabs(time_step) <= std::numeric_limits<double>::epsilon()) {
142  return StateMatrix::Identity();
143  }
144 
145  StateMatrix jacobian = StateMatrix::Zero();
146 
147  // Extract quaternion and normalize
148  Eigen::Quaterniond orientation(
149  current_state[SIdx::Quaternion::W],
150  current_state[SIdx::Quaternion::X],
151  current_state[SIdx::Quaternion::Y],
152  current_state[SIdx::Quaternion::Z]
153  );
154  orientation.normalize();
155 
156  // Get rotation matrix (world to body frame)
157  Matrix3d rotation_matrix_w_to_b = orientation.toRotationMatrix().transpose();
158 
159  // Position block - identity for position elements
160  jacobian.block<3,3>(SIdx::Position::Begin(), SIdx::Position::Begin()) =
161  Matrix3d::Identity();
162 
163  // Position wrt linear velocity
164  jacobian.block<3,3>(SIdx::Position::Begin(), SIdx::LinearVelocity::Begin()) =
165  rotation_matrix_w_to_b * (time_step * Matrix3d::Identity());
166 
167  // Position wrt linear acceleration
168  jacobian.block<3,3>(SIdx::Position::Begin(), SIdx::LinearAcceleration::Begin()) =
169  rotation_matrix_w_to_b * (time_step * time_step * 0.5 * Matrix3d::Identity());
170 
171  // Remove coupling between linear and angular position in the jacobian by
172  // setting to zero. Derivation of partial of position with respect to the
173  // rotation in quaternion form can be found here:
174  // http://www.iri.upc.edu/people/jsola/JoanSola/objectes/notes/kinematics.pdf
175  jacobian.block<3,4>(SIdx::Position::Begin(), SIdx::Quaternion::Begin()) =
176  Eigen::Matrix<double, 3, 4>::Zero();
177 
178  // Quaternion self-propagation
179  jacobian.block<4,4>(SIdx::Quaternion::Begin(), SIdx::Quaternion::Begin()) =
180  Matrix4d::Identity();
181 
182  // Extract angular velocity and acceleration components
183  const Vector3d angular_velocity =
184  current_state.segment<3>(SIdx::AngularVelocity::Begin());
185 
186  const Vector3d angular_acceleration =
187  current_state.segment<3>(SIdx::AngularAcceleration::Begin());
188 
189  // Calculate combined angular motion vector v = 0.5*ω*dt + 0.25*ω̇*dt²
190  // This is the same vector used in the quaternion update: q(t+dt) = exp(v) * q(t)
191  double t = time_step;
192  Vector3d combined_angular_motion = angular_velocity * t + 0.5 * angular_acceleration * t * t;
193  double motion_norm = combined_angular_motion.norm();
194 
195  // Jacobian terms for quaternion derivatives with respect to angular velocity
196  // ∂q/∂ω ≈ (t/2)*I - higher order correction terms
197  Matrix3d velocity_term = (t/2.0) * Matrix3d::Identity();
198 
199  // Jacobian terms for quaternion derivatives with respect to angular acceleration
200  // ∂q/∂ω̇ ≈ (t²/4)*I - higher order correction terms
201  Matrix3d accel_term = (t*t/4.0) * Matrix3d::Identity();
202 
203  // Apply correction terms when motion is non-zero
204  if (motion_norm > std::numeric_limits<double>::min()) {
205  // Compute outer product correction based on combined motion
206  // This comes from the derivative of the exponential map
207  Matrix3d outer_product_correction = (combined_angular_motion * combined_angular_motion.transpose()) /
208  (motion_norm * motion_norm);
209 
210  // Scale for velocity: (t³*||v||)/16 * (v*v^T/||v||²)
211  Matrix3d velocity_correction = (t*t*t * motion_norm / 16.0) * outer_product_correction;
212 
213  // Scale for acceleration: (t⁴*||v||)/32 * (v*v^T/||v||²)
214  Matrix3d accel_correction = (t*t*t*t * motion_norm / 32.0) * outer_product_correction;
215 
216  velocity_term -= velocity_correction;
217  accel_term -= accel_correction;
218  }
219 
220  // Create quaternion derivatives for angular velocity components
221  // Format: [scalar_part, vector_part_x, vector_part_y, vector_part_z]
222  Vector4d dq_dx(
223  -0.25 * t * combined_angular_motion[0], // Scalar part derivative
224  velocity_term(0,0), velocity_term(0,1), velocity_term(0,2) // Vector part derivatives
225  );
226  Vector4d dq_dy(
227  -0.25 * t * combined_angular_motion[1],
228  velocity_term(1,0), velocity_term(1,1), velocity_term(1,2)
229  );
230  Vector4d dq_dz(
231  -0.25 * t * combined_angular_motion[2],
232  velocity_term(2,0), velocity_term(2,1), velocity_term(2,2)
233  );
234 
235  // Set the Jacobian block for quaternion wrt angular velocity
236  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularVelocity::X) = dq_dx;
237  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularVelocity::Y) = dq_dy;
238  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularVelocity::Z) = dq_dz;
239 
240  // Create quaternion derivatives for angular acceleration components
241  // Format: [scalar_part, vector_part_x, vector_part_y, vector_part_z]
242  Vector4d dq_dax(
243  -0.125 * t * t * combined_angular_motion[0], // Scalar part derivative
244  accel_term(0,0), accel_term(0,1), accel_term(0,2) // Vector part derivatives
245  );
246  Vector4d dq_day(
247  -0.125 * t * t * combined_angular_motion[1],
248  accel_term(1,0), accel_term(1,1), accel_term(1,2)
249  );
250  Vector4d dq_daz(
251  -0.125 * t * t * combined_angular_motion[2],
252  accel_term(2,0), accel_term(2,1), accel_term(2,2)
253  );
254 
255  // Set the Jacobian block for quaternion wrt angular acceleration
256  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularAcceleration::X) = dq_dax;
257  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularAcceleration::Y) = dq_day;
258  jacobian.block<4,1>(SIdx::Quaternion::Begin(), SIdx::AngularAcceleration::Z) = dq_daz;
259 
260  // Velocity blocks - identity for velocity elements
261  jacobian.block<3,3>(SIdx::LinearVelocity::Begin(), SIdx::LinearVelocity::Begin()) =
262  Matrix3d::Identity();
263 
264  // Linear velocity wrt linear acceleration
265  jacobian.block<3,3>(SIdx::LinearVelocity::Begin(), SIdx::LinearAcceleration::Begin()) =
266  time_step * Matrix3d::Identity();
267 
268  // Angular velocity blocks - identity for angular velocity elements
269  jacobian.block<3,3>(SIdx::AngularVelocity::Begin(), SIdx::AngularVelocity::Begin()) =
270  Matrix3d::Identity();
271 
272  // Angular velocity wrt angular acceleration
273  jacobian.block<3,3>(SIdx::AngularVelocity::Begin(), SIdx::AngularAcceleration::Begin()) =
274  time_step * Matrix3d::Identity();
275 
276  // Acceleration blocks are modelled as zero (no change)
277 
278  // Velocity blocks - identity for velocity elements
279  jacobian.block<3,3>(SIdx::LinearVelocity::Begin(), SIdx::LinearVelocity::Begin()) =
280  Matrix3d::Identity();
281 
282  // Linear velocity wrt linear acceleration
283  jacobian.block<3,3>(SIdx::LinearVelocity::Begin(), SIdx::LinearAcceleration::Begin()) =
284  time_step * Matrix3d::Identity();
285 
286  // Angular velocity blocks - identity for angular velocity elements
287  jacobian.block<3,3>(SIdx::AngularVelocity::Begin(), SIdx::AngularVelocity::Begin()) =
288  Matrix3d::Identity();
289 
290  // Angular velocity wrt angular acceleration
291  jacobian.block<3,3>(SIdx::AngularVelocity::Begin(), SIdx::AngularAcceleration::Begin()) =
292  time_step * Matrix3d::Identity();
293 
294  jacobian.block<3,3>(SIdx::AngularAcceleration::Begin(), SIdx::AngularAcceleration::Begin()) =
295  std::exp(-kLambda * time_step) * Matrix3d::Identity();
296 
297  jacobian.block<3,3>(SIdx::LinearAcceleration::Begin(), SIdx::LinearAcceleration::Begin()) =
298  std::exp(-kLambda * time_step) * Matrix3d::Identity();
