220  bool lookupStaticTransform() {
221  if (sensor_frame_id_ == body_frame_id_) {
222  // Special case: sensor is in the body frame, use identity
223  sensor_to_body_transform_ = Eigen::Isometry3d::Identity();
224  filter_->SetSensorPoseInBodyFrameByIndex(sensor_index_, sensor_to_body_transform_);
225  has_valid_transform_ = true;
226  RCLCPP_INFO(node_->get_logger(), "Sensor %s is in body frame %s, using identity transform",
227  sensor_id_.c_str(), body_frame_id_.c_str());
228  return true;
229  }
230 
231  try {
232  // Try to look up the static transform
233  geometry_msgs::msg::TransformStamped transform_stamped;
234  transform_stamped = tf_buffer_->lookupTransform(
235  body_frame_id_, sensor_frame_id_, tf2::TimePointZero);
236 
237  // Convert to Eigen transform
238  Eigen::Isometry3d transform = Eigen::Isometry3d::Identity();
239  transform.translation() = Eigen::Vector3d(
240  transform_stamped.transform.translation.x,
241  transform_stamped.transform.translation.y,
242  transform_stamped.transform.translation.z);
243 
244  Eigen::Quaterniond q(
245  transform_stamped.transform.rotation.w,
246  transform_stamped.transform.rotation.x,
247  transform_stamped.transform.rotation.y,
248  transform_stamped.transform.rotation.z);
249  transform.linear() = q.toRotationMatrix();
250 
251  // Store and set transform
252  sensor_to_body_transform_ = transform;
253  filter_->SetSensorPoseInBodyFrameByIndex(sensor_index_, sensor_to_body_transform_);
254  has_valid_transform_ = true;
255 
256  RCLCPP_INFO(node_->get_logger(), "Set transform from %s to %s for sensor %s: [%f, %f, %f]",
257  sensor_frame_id_.c_str(), body_frame_id_.c_str(), sensor_id_.c_str(),
258  transform.translation().x(),
259  transform.translation().y(),
260  transform.translation().z());
261  return true;
262  } catch (const tf2::TransformException& ex) {
263  // Report the error without setting any transform
264  RCLCPP_WARN_THROTTLE(node_->get_logger(), *node_->get_clock(), 5000,
265  "Waiting for transform from %s to %s for sensor %s: %s",
266  sensor_frame_id_.c_str(), body_frame_id_.c_str(),

276  void messageCallback(const typename MsgType::SharedPtr msg) {
277  if(!time_manager_->isInitialized()) {
278  RCLCPP_WARN(node_->get_logger(), "Time manager not initialized for sensor %s", sensor_id_.c_str());
279  return;
280  }
281 
282  if (!msg) {
283  RCLCPP_WARN(node_->get_logger(), "Received null message for sensor %s", sensor_id_.c_str());
284  return;
285  }
286 
287  // Check for transform if we don't have a valid one yet
288  if (!has_valid_transform_) {
289  if (!lookupStaticTransform()) {
290  // Still don't have transform, skip processing this message
291  // Only log at most once per second to avoid spamming
292  RCLCPP_WARN_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
293  "Skipping measurement for sensor '%s' - waiting for transform from %s to %s",
294  sensor_name_.c_str(), sensor_frame_id_.c_str(), body_frame_id_.c_str());
295  return;
296  }
297  }
298 
299  // Convert to vector
300  DynamicVector measurement;
301  if (!msgToVector(*msg, measurement)) {
302  RCLCPP_WARN(node_->get_logger(), "Failed to convert message to measurement vector for sensor %s",
303  sensor_id_.c_str());
304  return;
305  }
306 
307  // Get timestamp from message using our offset-adjusted conversion
308  double timestamp = time_manager_->rosTimeToFilterTime(rclcpp::Time(msg->header.stamp));
309 
310  // Process the measurement
311  bool success = filter_->ProcessMeasurementByIndex(sensor_index_, measurement, timestamp);
312 
313  if (!success) {
314  RCLCPP_WARN(node_->get_logger(), "Failed to process measurement for sensor %s",
315  sensor_id_.c_str());
316  }
317 
318  // Always publish expected measurements regardless of process success
319  publishExpectedMeasurement(msg->header);
320  }

325  void publishExpectedMeasurement(const std_msgs::msg::Header& header) {
326  // Get the timestamp using our offset-adjusted conversion
327  double timestamp = time_manager_->rosTimeToFilterTime(rclcpp::Time(header.stamp));
328 
329  // Predict to the measurement time
330  auto state = filter_->GetStateEstimate(timestamp);
331 
332  // Get expected measurement for this sensor at this state
333  DynamicVector expected_vector;
334  if (!filter_->GetExpectedMeasurementByIndex(sensor_index_, expected_vector, state)) {
335  RCLCPP_WARN(node_->get_logger(), "Failed to get expected measurement for sensor %s",
336  sensor_id_.c_str());
337  return;
338  }
339 
340  // Get the measurement covariance
341  DynamicCovariance covariance;
342  if (!filter_->GetSensorCovarianceByIndex(sensor_index_, covariance)) {
343  RCLCPP_WARN(node_->get_logger(), "Failed to get covariance for sensor %s",
344  sensor_id_.c_str());
345  return;
346  }
347 
348  // Convert to message
349  auto expected_msg = vectorToMsg(expected_vector, header);
350 
351  // Apply bounds
352  auto upper_bound = applyBound(expected_msg, covariance, true);
353  auto lower_bound = applyBound(expected_msg, covariance, false);
354 
355  // Publish all three messages
356  expected_pub_->publish(expected_msg);
357  upper_bound_pub_->publish(upper_bound);
358  lower_bound_pub_->publish(lower_bound);
359  }
