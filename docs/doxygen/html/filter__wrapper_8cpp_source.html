169 geometry_msgs::msg::PoseWithCovarianceStamped FilterWrapper::getPoseEstimate() {
170 
171  double time_sec = filter_->GetCurrentTime();
172 
173  // Get state and covariance at specified time
174  auto state = filter_->GetStateEstimate(time_sec);
175  auto covariance = filter_->GetStateCovariance();
176 
177  // Create message
178  geometry_msgs::msg::PoseWithCovarianceStamped msg;
179  msg.header.stamp = time_manager_->filterTimeToRosTime(time_sec);
180  msg.header.frame_id = world_frame_id_;
181 
182  // Position
183  msg.pose.pose.position.x = state(StateIndex::Position::X);
184  msg.pose.pose.position.y = state(StateIndex::Position::Y);
185  msg.pose.pose.position.z = state(StateIndex::Position::Z);
186 
187  // Orientation (quaternion)
188  msg.pose.pose.orientation.w = state(StateIndex::Quaternion::W);
189  msg.pose.pose.orientation.x = state(StateIndex::Quaternion::X);
190  msg.pose.pose.orientation.y = state(StateIndex::Quaternion::Y);
191  msg.pose.pose.orientation.z = state(StateIndex::Quaternion::Z);
192 
193  // Fill covariance (position and orientation blocks)
194  for (int i = 0; i < 3; ++i) {
195  for (int j = 0; j < 3; ++j) {
196  // Position covariance (top-left 3x3 block)
197  msg.pose.covariance[i*6 + j] = covariance(StateIndex::Position::X + i, StateIndex::Position::X + j);
198 
199  // Orientation covariance (bottom-right 3x3 block)
200  // Note: This is an approximation as we're mapping quaternion to Euler angle representation
201  msg.pose.covariance[(i+3)*6 + (j+3)] = covariance(StateIndex::Quaternion::X + i, StateIndex::Quaternion::X + j);
202  }
203  }
204 
205  return msg;
206 }
