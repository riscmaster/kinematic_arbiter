81  void publish(
82  const MeasurementModelInterface::DynamicVector& measurement,
83  const rclcpp::Time& timestamp,
84  const std::string& frame_id,
85  std::mt19937& generator) {
86 
87  // Verify the dimension of the incoming measurement
88  int expected_dim = kinematic_arbiter::core::GetMeasurementDimension(sensor_type_);
89  if (measurement.size() != expected_dim) {
90  RCLCPP_ERROR(node_->get_logger(), "Measurement dimension mismatch for %s: expected %d, got %d",
91  name_.c_str(), expected_dim, (int)measurement.size());
92  return;
93  }
94 
95  // Create the true message
96  auto true_msg = eigenToRos(measurement, frame_id, timestamp);
97 
98  // Create noise vector with proper dimension
99  Eigen::MatrixXd noise_covariance = Eigen::MatrixXd::Identity(expected_dim, expected_dim) * (noise_sigma_ * noise_sigma_);
100  Eigen::VectorXd noise = kinematic_arbiter::utils::generateMultivariateNoise(noise_covariance, generator);
101 
102  // Create the noisy message
103  auto raw_msg = eigenToRos(measurement + noise, frame_id, timestamp);
104 
105  // Create bound messages
106  Eigen::VectorXd upper_bound = measurement;
107  Eigen::VectorXd lower_bound = measurement;
108 
109  // Add/subtract 3*sigma to each component
110  for (int i = 0; i < expected_dim; ++i) {
111  upper_bound(i) += 3.0 * noise_sigma_;
112  lower_bound(i) -= 3.0 * noise_sigma_;
113  }
114 
115  auto upper_msg = eigenToRos(upper_bound, frame_id, timestamp);
116  auto lower_msg = eigenToRos(lower_bound, frame_id, timestamp);
117 
118  // Publish all variants
119  truth_pub_->publish(true_msg);
120  raw_pub_->publish(raw_msg);
121  upper_bound_pub_->publish(upper_msg);
122  lower_bound_pub_->publish(lower_msg);
123  }

219  // Normalize

221  if (norm > 1e-10) {
222  msg.pose.orientation.w = qw / norm;
223  msg.pose.orientation.x = qx / norm;
224  msg.pose.orientation.y = qy / norm;
225  msg.pose.orientation.z = qz / norm;
226  } else {
227  // Default to identity if quaternion is too small
228  msg.pose.orientation.w = 1.0;
229  msg.pose.orientation.x = 0.0;
