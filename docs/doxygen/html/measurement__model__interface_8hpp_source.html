180  MeasurementAuxData ComputeAuxiliaryData(
181  const StateVector& state,
182  const StateCovariance& state_covariance,
183  const DynamicVector& measurement) const {
184  ValidateMeasurementSize(measurement);
185 
186  // Calculate innovation: ν = y_k - h(x_k)
187  DynamicVector innovation = measurement - PredictMeasurement(state);
188 
189  // Get measurement Jacobian: C_k
190  DynamicJacobian jacobian = GetMeasurementJacobian(state);
191 
192  // Calculate innovation covariance: S = C_k P_k C_k^T + R
193  DynamicCovariance innovation_covariance =
194  jacobian * state_covariance * jacobian.transpose() + measurement_covariance_;
195 
196  return MeasurementAuxData(innovation, jacobian, innovation_covariance);
197  }

205  bool ValidateAuxiliaryData(const MeasurementAuxData& aux_data) const {
206  // Check innovation and jacobian for NaN/Inf
207  if (!aux_data.innovation.allFinite() || !aux_data.jacobian.allFinite()) {
208  std::cerr << "Invalid innovation or jacobian from " << SensorTypeToString(type_) << " sensor" << std::endl;
209  return false;
210  }
211 
212  // Check innovation covariance for NaN/Inf
213  if (!aux_data.innovation_covariance.allFinite()) {
214  std::cerr << "Innovation covariance contains NaN/Inf from " << SensorTypeToString(type_) << " sensor" << std::endl;
215  return false;
216  }
217 
218  // Check if innovation covariance is well-conditioned
219  Eigen::JacobiSVD<DynamicCovariance> svd(aux_data.innovation_covariance);
220  double condition_number = svd.singularValues()(0) /
221  std::max(svd.singularValues()(svd.singularValues().size() - 1), 1e-12);
222 
223  const double kConditionThreshold = 1e12;
224  if (condition_number > kConditionThreshold) {
225  std::cerr << "Ill-conditioned innovation covariance (cond=" << condition_number
226  << ") from " << SensorTypeToString(type_) << " sensor" << std::endl;
227  return false;
228  }
229 
230  // Check if LDLT decomposition succeeds (positive definite check)
231  Eigen::LDLT<DynamicCovariance> ldlt(aux_data.innovation_covariance);
232  if (ldlt.info() != Eigen::Success) {
233  std::cerr << "Innovation covariance not positive definite from "
234  << SensorTypeToString(type_) << " sensor" << std::endl;
235  return false;
236  }
237 
238  // All checks passed
239  return true;
240  }

252  bool ValidateAndMediate(
253  const StateVector& state,
254  const StateCovariance& state_covariance,
255  const double& measurement_timestamp,
256  const DynamicVector& measurement,
257  MeasurementAuxData& aux_data) {
258 
259  previous_measurement_data_ = MeasurementData(
260  measurement_timestamp,
261  measurement,
262  measurement_covariance_
263  );
264 
265  // Compute auxiliary data
266  aux_data = ComputeAuxiliaryData(state, state_covariance, measurement);
267 
268  // Validate auxiliary data
269  if (!ValidateAuxiliaryData(aux_data)) {
270  return false;
271  }
272 
273  // Skip innovation test for forced accept
274  if (validation_params_.mediation_action == MediationAction::ForceAccept) {
275  UpdateCovariance(aux_data.innovation);
276  previous_measurement_data_.covariance = measurement_covariance_;
277  return true;
278  }
279 
280  // Mahalanobis distance: d = ν^T S^-1 ν
281  double chi_squared_term = aux_data.innovation.transpose() *
282  aux_data.innovation_covariance.ldlt().solve(
283  aux_data.innovation);
284 
285  // Determine threshold for chi-squared test
286  double threshold = utils::CalculateChiSquareCriticalValueNDof(
287  aux_data.innovation.rows()-1, validation_params_.confidence_level);
288 
289  // Check if measurement passes validation
290  if (chi_squared_term < threshold) {
291  UpdateCovariance(aux_data.innovation);
292  previous_measurement_data_.covariance = measurement_covariance_;
293  return true;
294  }
295 
296  // Measurement Apply Mediation Action
297  if (validation_params_.mediation_action == MediationAction::AdjustCovariance) {
298  // Scale covariance to make chi-squared test pass
299  double scale_factor = chi_squared_term / threshold;
300  aux_data.innovation_covariance *= scale_factor;
301  measurement_covariance_ = aux_data.innovation_covariance - aux_data.jacobian * state_covariance * aux_data.jacobian.transpose();
302  previous_measurement_data_.covariance = measurement_covariance_;
303  return true;
304  }
305 
306  // Log warning with metadata when validation fails
307  std::cerr << "WARNING: Measurement validation failed for sensor type "
308  << SensorTypeToString(type_) << std::endl
309  << " Chi-squared value: " << chi_squared_term
310  << ", Threshold: " << threshold << std::endl
311  << " Innovation norm: " << aux_data.innovation.norm()
312  << ", Measurement timestamp: " << measurement_timestamp << std::endl;
313 
314  // Always return false if validation fails
315  return false;
316  }

456  void UpdateCovariance(const DynamicVector& innovation) {
457  // Define bounds as named constants for clarity
458  static const double kMinInnovation = 1e-6;
459  static const double kMaxInnovation = 1e6;
460 
461  // 1. Clip large values (positive and negative)
462  auto clipped = innovation.array().cwiseMax(-kMaxInnovation).cwiseMin(kMaxInnovation);
463 
464  // 2. Apply minimum magnitude while preserving sign
465  // Formula: sign(x) * max(|x|, kMinInnovation)
466  DynamicVector bounded_innovation =
467  clipped.sign() * clipped.abs().cwiseMax(kMinInnovation);
468 
469  // Update covariance with the adaptive formula
470  measurement_covariance_ += (bounded_innovation * bounded_innovation.transpose() -
471  measurement_covariance_) / validation_params_.noise_sample_window;
472  }
