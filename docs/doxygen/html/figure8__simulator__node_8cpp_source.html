166  void Figure8SimulatorNode::update() {
167  iteration_++;
168 
169  // Calculate current time and state
170  double elapsed_seconds = (this->now() - start_time_).seconds();
171  auto state = kinematic_arbiter::utils::Figure8Trajectory(elapsed_seconds, trajectory_config_);
172 
173  // Always publish ground truth and TF
174  publishGroundTruth(state);
175 
176  // Publish sensor data at their respective rates
177  if (iteration_ % position_divider_ == 0) {
178  publishPosition(state, elapsed_seconds);
179  }
180 
181  if (iteration_ % pose_divider_ == 0) {
182  publishPose(state, elapsed_seconds);
183  }
184 
185  if (iteration_ % velocity_divider_ == 0) {
186  publishVelocity(state, elapsed_seconds);
187  }
188 
189  if (iteration_ % imu_divider_ == 0) {
190  publishImu(state, elapsed_seconds);
191  }
192  }

194  void Figure8SimulatorNode::publishGroundTruth(const StateVector& state) {
195  auto current_time = this->now();
196 
197  // Extract components for convenience
198  Eigen::Vector3d position = state.segment<3>(SIdx::Position::Begin());
199  Eigen::Quaterniond quaternion(

203  state[SIdx::Quaternion::Z]);

205  Eigen::Vector3d angular_velocity = state.segment<3>(SIdx::AngularVelocity::Begin());
206

211  pose_msg.pose.position.x = position.x();

213  pose_msg.pose.position.z = position.z();
214  pose_msg.pose.orientation.w = quaternion.w();
215  pose_msg.pose.orientation.x = quaternion.x();
216  pose_msg.pose.orientation.y = quaternion.y();
217  pose_msg.pose.orientation.z = quaternion.z();
218  truth_pose_pub_->publish(pose_msg);
219 
220  // Publish truth velocity
221  auto vel_msg = geometry_msgs::msg::TwistStamped();
222  vel_msg.header.stamp = current_time;
223  vel_msg.header.frame_id = body_frame_id_;
224  vel_msg.twist.linear.x = linear_velocity.x();

226  vel_msg.twist.linear.z = linear_velocity.z();

228  vel_msg.twist.angular.y = angular_velocity.y();
229  vel_msg.twist.angular.z = angular_velocity.z();
230  truth_velocity_pub_->publish(vel_msg);
231 
232  // Publish transform
233  geometry_msgs::msg::TransformStamped transform;
234  transform.header.stamp = current_time;
235  transform.header.frame_id = world_frame_id_;
236  transform.child_frame_id = body_frame_id_;
237  transform.transform.translation.x = position.x();
238  transform.transform.translation.y = position.y();
239  transform.transform.translation.z = position.z();
240  transform.transform.rotation = pose_msg.pose.orientation;
241  tf_broadcaster_->sendTransform(transform);
