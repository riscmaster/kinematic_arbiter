197 template <typename Generator, typename Derived>
198 Eigen::VectorXd generateMultivariateNoise(
199  const Eigen::MatrixBase<Derived>& covariance,
200  Generator& generator) {
201 
202  const int size = covariance.rows();
203 
204  // Verify the covariance matrix is square
205  if (covariance.rows() != covariance.cols()) {
206  throw std::invalid_argument("Covariance matrix must be square");
207  }
208 
209  // Compute the Cholesky decomposition
210  Eigen::LLT<Eigen::MatrixXd> llt(covariance);
211 
212  // Check if the covariance matrix is positive definite
213  if (llt.info() != Eigen::Success) {
214  throw std::invalid_argument("Covariance matrix must be positive definite");
215  }
216 
217  // Get the lower triangular matrix L from the decomposition
218  Eigen::MatrixXd L = llt.matrixL();
219 
220  // Create standard normal distribution
221  std::normal_distribution<double> normal_dist(0.0, 1.0);
222 
223  // Generate standard normal samples
224  Eigen::VectorXd z(size);
225  for (int i = 0; i < size; ++i) {
226  z(i) = normal_dist(generator);
227  }
228 
229  // Transform to desired covariance: x = L*z
230  return L * z;
231 }
